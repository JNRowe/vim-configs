Add map to close wordnet_ window from anywhere::

    function WordnetCloseWin()
        if bufnr("__WordNet__") > -1
            exec bufnr("__WordNet__") . "bdelete!"
        endif
    endf

    nnoremap <silent> <Leader>wnq :call WordnetCloseWin()<CR>
